N=6144;
Fp=1/64*2*0.4535;
Fst=1/64*2*0.5465;
Ap=0.01;
Ast=190;

f=fdesign.lowpass( 'N,Fp,Ap,Ast', N, Fp, Ap, Ast );
Hf=design(f,'equiripple');
fvtool(Hf);

He = Hf;
He.Arithmetic = 'fixed';
set(He, 'CoeffWordLength', 28);
fvtool(He);
info(He);

fcfwrite(He, '../../rtl/rom/rom_src_6144x28b_cand.txt', 'hex');

Hd = Hf;
Hd.Arithmetic = 'fixed';
set(Hd, 'CoeffWordLength', 32);
fvtool(Hd);
info(Hd);

fcfwrite(Hd, '../../rtl/rom/rom_src_6144x32b_cand.txt', 'hex');
